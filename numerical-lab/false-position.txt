#include <iostream>
#include <cmath>
using namespace std;

double f(double x) {
    return x*x*x - x*x - 1;
}

int main() {
    double a = 1, b = 2, c;
    double eps = 0.0005;   // for 3 decimal places

    if (f(a) * f(b) >= 0) {
        cout << "Invalid interval";
        return 0;
    }

    do {
        c = (a*f(b) - b*f(a)) / (f(b) - f(a));

        if (f(a) * f(c) < 0)
            b = c;
        else
            a = c;

    } while (fabs(f(c)) > eps);

    cout << "Root = " << c << endl;
    return 0;
}
